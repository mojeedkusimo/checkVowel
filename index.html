function checkVowel(myString) {
            let a = 0;
            let e = 0;
            let i = 0;
            let o = 0;
            let u = 0;

            let x = 0;
            while (x < myString.length) {

                let lowerCase = myString.toLowerCase();
                switch (lowerCase.charAt(x)) {
                    case 'a':
                        a += 1;
                        break;
                    case 'e':
                        e += 1;
                        break;
                    case 'o':
                        o += 1;
                        break;
                    case 'u':
                        u += 1;
                        break;
                    case 'i':
                        i += 1;
                        break;
                }
              x++;
            }

            alert('a is: '+ a + ' & e is: '+ e + ' & i is: '+ i + ' & o is: '+ o + ' & u is: '+ u);
        }
